export default function SquaredCarousel2() {
  const [currentSlide, setCurrentSlide] = useState(0);
  const [sliderRef, instanceRef] = useKeenSlider({
    loop: true,
    slides: { perView: 2, spacing: 12 },
    slideChanged(slider) {
      setCurrentSlide(slider.track.details.rel);
    },
  });

  return (
    <div className="relative w-full max-w-7xl mx-auto group py-10">
      <div className="space-y-3 pb-20">
        <button className="bg-[#BAB2DC] text-[#0A0A0A] font-semibold py-1 px-6 rounded-full mb-4">
          Testimonials
        </button>
        <p className="text-5xl">
          Some kind words from our{" "}
          <span className="text-gray-400">Members</span>
        </p>
      </div>

      {/* Navigation Arrows */}
      <button
        className="absolute left-[62rem]  top-[8rem] -translate-y-1/2 z-10 p-3 rounded-xl bg-black/80 text-white hover:bg-black transition-all opacity-100  shadow-lg hover:scale-105"
        onClick={() => instanceRef.current?.prev()}
        aria-label="Previous slide"
      >
        <ArrowLeft className="w-6 h-6" />
      </button>
      <button
        className="absolute right-[10rem] top-[8rem]  -translate-y-1/2 z-10 p-3 rounded-xl bg-black/80 text-white hover:bg-black transition-all opacity-100 shadow-lg hover:scale-105"
        onClick={() => instanceRef.current?.next()}
        aria-label="Next slide"
      >
        <ArrowRight className="w-6 h-6" />
      </button>

      {/* Slider */}
      <div ref={sliderRef} className="keen-slider">
        {testimonials.map((testimonial, index) => (
          <div
            key={index}
            className="keen-slider__slide bg-white p-8 flex flex-col items-center text-center"
          >
            <div className="mt-11">
            <BiSolidQuoteLeft className="h-12 w-12 text-gray-500" />
            </div>
            <p className="text-2xl italic max-w-sm pt-8">“{testimonial.text}”</p>
            <h3 className="text- font-semibold mt-8">{testimonial.author}</h3>
            <p className="text-[1rem] text-gray-400">{testimonial.company}</p>
          </div>
        ))}
      </div>

      {/* Navigation Dots */}
      <div className="flex justify-center md:hidden gap-2 mt-12">
        {testimonials.map((_, idx) => (
          <button
            key={idx}
            className={`w-3 h-1 rounded-full transition-all ${
              currentSlide === idx ? "bg-gray-900 w-6" : "bg-gray-300"
            }`}
            onClick={() => instanceRef.current?.moveToIdx(idx)}
            aria-label={Go to slide ${idx + 1}}
          />
        ))}
      </div>
    </div>
  );
}
